\begin{frame}{Training and Performance Metrics}

Pixel Accuracy: 

\begin{equation*}
PA = \frac{\sum TP+ \epsilon}{\sum TP +\sum FP}
\end{equation*}

DICE Metric: 

\begin{equation*}
DICE = \frac{TP}{2TP+FP+FN}
\end{equation*}

Precision:
\begin{equation*}
P = \frac{TP}{TP+FP}
\end{equation*}

Recall:
\begin{equation*}
R = \frac{TP}{TP+FN}
\end{equation*}

\end{frame}
